  <script>
    // State
    let expr = '';          // raw expression string
    let justCalculated = false;

    const exprEl   = document.getElementById('exprEl');
    const resultEl = document.getElementById('resultEl');
    const display  = document.getElementById('display');

    // Dark mode
    document.getElementById('darkToggle').addEventListener('change', function () {
      document.body.classList.toggle('dark', this.checked);
    });

    // -- Helpers --
    function updateDisplay(exprStr, resultStr = '', isError = false) {
      exprEl.textContent   = exprStr || '0';
      resultEl.textContent = resultStr;
      display.classList.toggle('error', isError);
    }

    function liveResult() {
      if (!expr) { resultEl.textContent = ''; return; }
      try {
        const val = evalExpr(expr);
        if (!isNaN(val) && isFinite(val)) {
          resultEl.textContent = '= ' + formatNum(val);
        } else {
          resultEl.textContent = '';
        }
      } catch {
        resultEl.textContent = '';
      }
    }

    function formatNum(n) {
      const s = parseFloat(n.toFixed(10)).toString();
      return s.replace(/\B(?=(\d{3})+(?!\d))/g, ',');
    }

    // Evaluate expression (replace display chars with JS operators)
    function evalExpr(str) {
      // Replace % with /100 context-aware
      let s = str
        .replace(/÷/g, '/')
        .replace(/×/g, '*')
        .replace(/−/g, '-')
        .replace(/(\d+\.?\d*)\s*%/g, '($1/100)');

      // Basic validation: must start with digit or minus or (
      if (!/^[-\d(]/.test(s)) throw new Error('invalid');

      // eslint-disable-next-line no-new-func
      return Function('"use strict"; return (' + s + ')')();
    }

    // -- Input handlers --
    function inputChar(c) {
      // If just calculated and user presses digit/dot, start fresh
      if (justCalculated && /[\d.]/.test(c)) { expr = ''; }
      justCalculated = false;

      // Prevent multiple dots in current number segment
      if (c === '.') {
        const lastNum = expr.split(/[÷×+−%]/).pop();
        if (lastNum.includes('.')) return;
        if (lastNum === '') expr += '0'; // leading dot → 0.
      }

      // Prevent leading zeros
      if (c === '0' && expr === '0') return;

      expr += c;
      updateDisplay(expr);
      liveResult();
    }

    function inputOp(op) {
      justCalculated = false;
      if (!expr && op !== '−') return; // allow leading minus

      // Replace last operator if expr ends with one
      const lastChar = expr.slice(-1);
      if (['÷','×','+','−'].includes(lastChar)) {
        expr = expr.slice(0, -1);
      }

      // Don't allow operator after %
      if (expr.slice(-1) === '%' && op === '%') return;

      expr += op;
      updateDisplay(expr);
      liveResult();
    }

    function clearAll() {
      expr = '';
      justCalculated = false;
      updateDisplay('0');
      resultEl.textContent = '';
    }

    function clearHistory() { clearAll(); }

    function backspace() {
      if (justCalculated) { clearAll(); return; }
      expr = expr.slice(0, -1);
      updateDisplay(expr || '0');
      liveResult();
    }

    function toggleSign() {
      if (!expr || expr === '0') return;
      // Wrap expression in negation
      if (expr.startsWith('-(') && expr.endsWith(')')) {
        expr = expr.slice(2, -1);
      } else {
        expr = '-(' + expr + ')';
      }
      updateDisplay(expr);
      liveResult();
    }

    function calculate() {
      // Validation: must not be empty
      if (!expr || expr === '0') return;

      // Trailing operator check
      const lastChar = expr.slice(-1);
      if (['÷','×','+','−'].includes(lastChar)) {
        updateDisplay(expr, 'Input tidak lengkap', true);
        return;
      }

      try {
        const val = evalExpr(expr);
        if (!isFinite(val)) {
          updateDisplay(expr, 'Tidak dapat dibagi nol', true);
          return;
        }
        const formatted = formatNum(val);
        exprEl.textContent = expr + ' =';
        resultEl.textContent = formatted;
        resultEl.classList.remove('pop');
        void resultEl.offsetWidth; // reflow
        resultEl.classList.add('pop');
        expr = String(parseFloat(val.toFixed(10)));
        justCalculated = true;
        display.classList.remove('error');
      } catch {
        updateDisplay(expr, 'Input tidak valid', true);
      }
    }

    // Keyboard support
    document.addEventListener('keydown', (e) => {
      if (e.key >= '0' && e.key <= '9') inputChar(e.key);
      else if (e.key === '.') inputChar('.');
      else if (e.key === '+') inputOp('+');
      else if (e.key === '-') inputOp('−');
      else if (e.key === '*') inputOp('×');
      else if (e.key === '/') { e.preventDefault(); inputOp('÷'); }
      else if (e.key === '%') inputChar('%');
      else if (e.key === 'Enter' || e.key === '=') calculate();
      else if (e.key === 'Backspace') backspace();
      else if (e.key === 'Escape') clearAll();
    });

    // Init
    updateDisplay('0');
  </script>
